OOP Inheritance
คือการสืบทอด object นึงไปอีก object นึงเช่น
class Chain {
    constructor (chain,kin){
        this.chain = chain
        this.kin = kin
    }
    getinfo() {
        return this.chain + " " + this.kin
    }
}
class Getter extends Chain {  //อย่างนี้คือการอ้างถึง class Chain
    constructor(chain,kin,getter){
    super(chain,kin)
    this.getter = getter
    }
    summary(){
        this.getter + this.chain + this.kin
    }
}
let k = new Getter("getter","chain","kin")
